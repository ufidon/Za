\usepackage{fp} % For floating-point arithmetic

% Define the emoji font (e.g., Symbola or Noto Color Emoji)
\newfontfamily\emojifont{Noto Color Emoji}[Renderer=Harfbuzz]

% Define the \emojitext command with dynamic baseline skip
\newcommand{\emojitext}[3]{%
  \FPmul{\行距}{1.2}{#1}% Multiply fontsizenumber by 1.2
  {\fontsize{#1#2}{\行距#2}\selectfont\emojifont #3}%
}
